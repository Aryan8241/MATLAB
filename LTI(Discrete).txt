clc;
clear all;
close all;

% Discrete signals
x = [1 2 3 4];
h = [1 1 1];

% Convolution
y = conv(x, h);

% Indexing
n_x = 0:length(x)-1;
n_h = 0:length(h)-1;
n_y = 0:length(y)-1;

% Plotting
subplot(3,1,1)
stem(n_x, x)
title('Input Signal x[n]')
xlabel('n'); ylabel('x[n]')

subplot(3,1,2)
stem(n_h, h)
title('Impulse Response h[n]')
xlabel('n'); ylabel('h[n]')

subplot(3,1,3)
stem(n_y, y)
title('Output Signal y[n]')
xlabel('n'); ylabel('y[n]')
